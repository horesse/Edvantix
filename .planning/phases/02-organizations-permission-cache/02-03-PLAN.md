    2. Create **PermissionRequirementHandler.cs**:
       ```csharp
       using Grpc.Core;
       using Microsoft.AspNetCore.Authorization;
       using Microsoft.Extensions.Logging;

       namespace Edvantix.Chassis.Security.Authorization;

       /// <summary>
       /// Resolves permission checks by calling Organizations gRPC CheckPermission endpoint.
       /// Extracts userId from "profile_id" claim and schoolId from "X-School-Id" header (via ITenantContext or HttpContext).
       /// </summary>
       internal sealed class PermissionRequirementHandler(
           PermissionsGrpcService.PermissionsGrpcServiceClient grpcClient,
           ILogger<PermissionRequirementHandler> logger
       ) : AuthorizationHandler<PermissionRequirement>
       {
           protected override async Task HandleRequirementAsync(
               AuthorizationHandlerContext context,
               PermissionRequirement requirement)
           {
               // Extract profile_id from claims
               var profileIdClaim = context.User.FindFirst("profile_id");
               if (profileIdClaim is null || !Guid.TryParse(profileIdClaim.Value, out var profileId))
               {
                   logger.LogWarning("Permission check failed: no profile_id claim");
                   return; // Fail — don't call context.Succeed
               }

               // Extract school_id from X-School-Id header via claims or HttpContext
               // The TenantMiddleware adds the school_id as a claim or we read from HttpContext
               var schoolIdClaim = context.User.FindFirst("school_id");
               if (schoolIdClaim is null || !Guid.TryParse(schoolIdClaim.Value, out var schoolId))
               {
                   // Fallback: try to get from HttpContext if available
                   if (context.Resource is Microsoft.AspNetCore.Http.HttpContext httpContext)
                   {
                       var header = httpContext.Request.Headers["X-School-Id"].FirstOrDefault();
                       if (header is null || !Guid.TryParse(header, out schoolId))
                       {
                           logger.LogWarning("Permission check failed: no school_id available");
                           return;
                       }
                   }
                   else
                   {
                       return;
                   }
               }

               try
               {
                   var reply = await grpcClient.CheckPermissionAsync(
                       new CheckPermissionRequest
                       {
                           UserId = profileId.ToString(),
                           SchoolId = schoolId.ToString(),
                           Permission = requirement.Permission,
                       });

                   if (reply.HasPermission)
                   {
                       context.Succeed(requirement);
                   }
               }
               catch (RpcException ex)
               {
                   logger.LogError(ex, "gRPC CheckPermission failed for {ProfileId}/{SchoolId}/{Permission}",
                       profileId, schoolId, requirement.Permission);
                   // Fail authorization on gRPC errors — do not succeed
               }
           }
       }
       ```
       IMPORTANT: The handler must reference the proto-generated client type. The Chassis project needs a Protobuf reference to the Organizations proto with `GrpcServices="Client"`. Copy the proto file to Chassis or reference it from Organizations. Follow the same pattern as Organizations copies Persona's proto.

       Since Chassis is a shared library, the best approach is:
       - Copy `permissions.proto` to `src/BuildingBlocks/Edvantix.Chassis/Security/Protos/organizations/v1/permissions.proto`
       - Add `<Protobuf Include="Security\Protos\organizations\v1\permissions.proto" GrpcServices="Client" />` to `Edvantix.Chassis.csproj`
       - Ensure Chassis has `Grpc.Tools` and `Google.Protobuf` + `Grpc.Net.Client` package references

       Alternatively, if Chassis already depends on gRPC packages, just add the proto. Check Chassis.csproj first.

       If Chassis does NOT have gRPC dependencies and adding them is too heavyweight, then:
       - Create the PermissionRequirementHandler to accept an `IPermissionChecker` interface (defined in Chassis)
       - Put the gRPC implementation in a separate extension method that downstream services call

    3. Add to **PolicyBuilderExtensions.cs**:
       ```csharp
       /// <summary>
       /// Adds a permission requirement to the authorization policy.
       /// Use with <see cref="AddPermissionAuthorization"/> in service registration.
       /// </summary>
       public static AuthorizationPolicyBuilder RequirePermission(
           this AuthorizationPolicyBuilder builder,
           string permission
       ) => builder.AddRequirements(new PermissionRequirement(permission));

       /// <summary>
       /// Registers the Organizations gRPC client and <see cref="PermissionRequirementHandler"/>
       /// for permission-based authorization. Call this in downstream services (Scheduling, Payments).
       /// </summary>
       /// <param name="builder">The host application builder.</param>
       /// <param name="organizationsGrpcBaseUrl">The base URL for the Organizations gRPC service.</param>
       public static IServiceCollection AddPermissionAuthorization(
           this IHostApplicationBuilder builder,
           string organizationsGrpcBaseUrl)
       {
           builder.Services.AddGrpcClient<PermissionsGrpcService.PermissionsGrpcServiceClient>(
               o => o.Address = new Uri(organizationsGrpcBaseUrl));
           builder.Services.AddSingleton<IAuthorizationHandler, PermissionRequirementHandler>();
           return builder.Services;
       }
       ```

       NOTE: The `AddPermissionAuthorization` method takes `IHostApplicationBuilder` (not just `IServiceCollection`) to match the existing pattern. If Aspire service discovery is used, downstream services will pass the Aspire-resolved URL. If `AddGrpcServiceReference` is available from Aspire, prefer that over manual `AddGrpcClient`.
