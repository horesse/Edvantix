namespace Edvantix.Organizations.Features.Roles.AssignPermissions;

public sealed class AssignPermissionsCommand : ICommand
{
    public required Guid RoleId { get; init; }
    public required List<string> PermissionNames { get; init; }
}

public sealed class AssignPermissionsCommandHandler(
    IRoleRepository roleRepository,
    IPermissionRepository permissionRepository
) : ICommandHandler<AssignPermissionsCommand>
{
    public async ValueTask<Unit> Handle(
        AssignPermissionsCommand request, CancellationToken cancellationToken)
    {
        var role = await roleRepository.FindByIdAsync(request.RoleId, cancellationToken)
            ?? throw new NotFoundException(nameof(Role), request.RoleId);

        if (request.PermissionNames.Count == 0)
        {
            role.SetPermissions([]);
            await roleRepository.UnitOfWork.SaveEntitiesAsync(cancellationToken);
            return Unit.Value;
        }

        // Validate all permission names exist in the catalogue
        var permissions = await permissionRepository.GetByNamesAsync(
            request.PermissionNames, cancellationToken);

        var foundNames = permissions.Select(p => p.Name).ToHashSet();
        var missingNames = request.PermissionNames.Where(n => !foundNames.Contains(n)).ToList();

        if (missingNames.Count > 0)
            throw new InvalidOperationException(
                $"Unknown permission(s): {string.Join(", ", missingNames)}. " +
                "Register them first via the permission registration endpoint.");

        role.SetPermissions(permissions.Select(p => p.Id));
        await roleRepository.UnitOfWork.SaveEntitiesAsync(cancellationToken);
        return Unit.Value;
    }
}
```

namespace Edvantix.Organizations.Features.Roles.GetRolePermissions;

public sealed record RolePermissionItem(Guid Id, string Name);

public sealed class GetRolePermissionsQuery : IQuery<List<RolePermissionItem>>
{
    public required Guid RoleId { get; init; }
}

public sealed class GetRolePermissionsQueryHandler(
    IRoleRepository roleRepository,
    IPermissionRepository permissionRepository
) : IQueryHandler<GetRolePermissionsQuery, List<RolePermissionItem>>
{
    public async ValueTask<List<RolePermissionItem>> Handle(
        GetRolePermissionsQuery request, CancellationToken cancellationToken)
    {
        var role = await roleRepository.FindByIdAsync(request.RoleId, cancellationToken)
            ?? throw new NotFoundException(nameof(Role), request.RoleId);

        var permissionIds = role.Permissions.Select(rp => rp.PermissionId).ToList();
        if (permissionIds.Count == 0) return [];

        // Fetch Permission entities to get names
        var allPermissions = await permissionRepository.GetAllAsync(cancellationToken);
        return allPermissions
            .Where(p => permissionIds.Contains(p.Id))
            .Select(p => new RolePermissionItem(p.Id, p.Name))
            .OrderBy(p => p.Name)
            .ToList();
    }
}
```

Note: This implementation fetches all permissions and filters in memory. This is acceptable for v1 since the Permission catalogue will be small (tens of entries). If needed later, add `GetByIdsAsync` to `IPermissionRepository`.
