```csharp
namespace Edvantix.Chassis.Security.Tenant;

public sealed class TenantContext : ITenantContext
{
    private Guid _schoolId;

    public Guid SchoolId => IsResolved
        ? _schoolId
        : throw new InvalidOperationException("Tenant context not resolved. Ensure X-School-Id header is present.");

    public bool IsResolved { get; private set; }

    public void Resolve(Guid schoolId)
    {
        _schoolId = schoolId;
        IsResolved = true;
    }
}
```

File `src/BuildingBlocks/Edvantix.Chassis/Security/Tenant/TenantMiddleware.cs`:
```csharp
namespace Edvantix.Chassis.Security.Tenant;

public sealed class TenantMiddleware(RequestDelegate next)
{
    private const string SchoolIdHeader = "X-School-Id";

    public async Task InvokeAsync(HttpContext context, ITenantContext tenantContext)
    {
        if (context.Request.Headers.TryGetValue(SchoolIdHeader, out var value)
            && Guid.TryParse(value, out var schoolId))
        {
            tenantContext.Resolve(schoolId);
        }

        await next(context);
    }
}
```

Create `Infrastructure/OrganizationsDbContext.cs`:
```csharp
using Edvantix.Organizations.Domain.AggregatesModel.PermissionAggregate;
using Edvantix.Organizations.Domain.AggregatesModel.RoleAggregate;
using Edvantix.Organizations.Domain.AggregatesModel.UserRoleAssignmentAggregate;

namespace Edvantix.Organizations.Infrastructure;

public sealed class OrganizationsDbContext(
    DbContextOptions<OrganizationsDbContext> options,
    ITenantContext tenantContext
) : DbContext(options), IUnitOfWork
{
    public DbSet<Role> Roles => Set<Role>();
    public DbSet<Permission> Permissions => Set<Permission>();
    public DbSet<UserRoleAssignment> UserRoleAssignments => Set<UserRoleAssignment>();
    public DbSet<RolePermission> RolePermissions => Set<RolePermission>();

    protected override void OnModelCreating(ModelBuilder modelBuilder)
    {
        base.OnModelCreating(modelBuilder);
        modelBuilder.AddInboxStateEntity();
        modelBuilder.AddOutboxMessageEntity();
        modelBuilder.AddOutboxStateEntity();
        modelBuilder.ApplyConfigurationsFromAssembly(typeof(OrganizationsDbContext).Assembly);

        // Tenant isolation query filters -- set here because ApplyConfigurationsFromAssembly
        // cannot inject ITenantContext into configuration constructors.
        // CRITICAL: For Role, combine tenant AND soft-delete in ONE HasQueryFilter call.
        // EF Core only supports one HasQueryFilter per entity -- second call overwrites first.
        modelBuilder.Entity<Role>().HasQueryFilter(r =>
            r.SchoolId == tenantContext.SchoolId && !r.IsDeleted);

        modelBuilder.Entity<UserRoleAssignment>().HasQueryFilter(a =>
            a.SchoolId == tenantContext.SchoolId);

        // Permission: NO query filter -- global catalogue
    }

    public async Task<bool> SaveEntitiesAsync(CancellationToken cancellationToken = default)
    {
        await SaveChangesAsync(cancellationToken);
        return true;
    }
}
```

This test uses the EF Core model API (not ArchUnitNET IL scanning) to verify that every entity implementing `ITenanted` has a registered `HasQueryFilter`. It instantiates `OrganizationsDbContext` with an in-memory provider and inspects the model.

```csharp
using Edvantix.ArchTests.Abstractions;
using Edvantix.Chassis.Security.Tenant;
using Edvantix.Organizations.Domain.Abstractions;
using Edvantix.Organizations.Infrastructure;
using Microsoft.EntityFrameworkCore;

namespace Edvantix.ArchTests.Domain;

public sealed class TenantIsolationTests : BaseTest
{
    [Test]
    public void GivenTenantedEntities_WhenCheckingOrganizationsDbContextModel_ThenAllShouldHaveQueryFilter()
    {
        // Arrange: build OrganizationsDbContext with in-memory provider + stub tenant context
        var tenantContext = new TenantContext();
        tenantContext.Resolve(Guid.NewGuid()); // resolve with any GUID so filters can be built

        var options = new DbContextOptionsBuilder<OrganizationsDbContext>()
            .UseInMemoryDatabase(databaseName: $"TenantIsolationTest_{Guid.NewGuid()}")
            .Options;

        using var dbContext = new OrganizationsDbContext(options, tenantContext);

        // Act: get all entity types that implement ITenanted
        var tenantedEntityTypes = dbContext.Model.GetEntityTypes()
            .Where(et => et.ClrType.IsAssignableTo(typeof(ITenanted)))
            .ToList();

        // Assert: each tenanted entity must have a query filter
        tenantedEntityTypes.Count.ShouldBeGreaterThan(0,
            "Expected at least one ITenanted entity in OrganizationsDbContext");

        foreach (var entityType in tenantedEntityTypes)
        {
            var queryFilter = entityType.GetQueryFilter();
            queryFilter.ShouldNotBeNull(
                $"Entity {entityType.ClrType.Name} implements ITenanted but has no HasQueryFilter registered. " +
                "Add HasQueryFilter in OrganizationsDbContext.OnModelCreating.");

            // Verify the filter expression contains SchoolId reference
            var filterString = queryFilter.ToString();
            filterString.ShouldContain("SchoolId",
                $"Entity {entityType.ClrType.Name} has a query filter but it does not reference SchoolId.");
        }
    }

    [Test]
    public void GivenPermissionEntity_WhenCheckingModel_ThenShouldNotHaveQueryFilter()
    {
        // Permission is a global catalogue -- it must NOT have a tenant query filter
        var tenantContext = new TenantContext();
        tenantContext.Resolve(Guid.NewGuid());

        var options = new DbContextOptionsBuilder<OrganizationsDbContext>()
            .UseInMemoryDatabase(databaseName: $"PermissionFilterTest_{Guid.NewGuid()}")
            .Options;

        using var dbContext = new OrganizationsDbContext(options, tenantContext);

        var permissionEntityType = dbContext.Model.FindEntityType(
            typeof(Edvantix.Organizations.Domain.AggregatesModel.PermissionAggregate.Permission));

        permissionEntityType.ShouldNotBeNull();
        permissionEntityType.GetQueryFilter().ShouldBeNull(
            "Permission is a global catalogue and must NOT have a tenant query filter.");
    }
}
```
