   ```csharp
   public sealed class AttendanceRecordedIntegrationEventHandler(
       PaymentsDbContext dbContext,
       ILogger<AttendanceRecordedIntegrationEventHandler> logger
   ) : IConsumer<AttendanceRecordedIntegrationEvent>
   {
       public async Task Consume(ConsumeContext<AttendanceRecordedIntegrationEvent> context)
       {
           var msg = context.Message;
           var processed = await dbContext.ProcessedAttendanceEvents
               .FirstOrDefaultAsync(p => p.CorrelationId == msg.CorrelationId,
                   context.CancellationToken);

           var lastStatus = processed?.LastStatus;
           var newStatus = msg.Status;

           // D-11 state machine:
           LessonTransactionType? transactionType = (lastStatus, newStatus) switch
           {
               (null, "Present") => LessonTransactionType.AttendanceDebit,
               (not "Present", "Present") => LessonTransactionType.AttendanceDebit,
               ("Present", not "Present") => LessonTransactionType.AttendanceReversal,
               ("Present", "Present") => null, // no-op
               _ => null // non-Present to non-Present (e.g. Absent to Excused) — no financial impact
           };

           if (transactionType is not null)
           {
               // Load or create StudentBalance — use IgnoreQueryFilters because
               // consumer runs outside HTTP context (no tenant context available)
               var balance = await dbContext.StudentBalances
                   .IgnoreQueryFilters()
                   .Include(b => b.Transactions)
                   .FirstOrDefaultAsync(b =>
                       b.StudentId == msg.StudentId && b.SchoolId == msg.SchoolId,
                       context.CancellationToken);

               if (balance is null)
               {
                   balance = new StudentBalance(msg.SchoolId, msg.StudentId);
                   dbContext.StudentBalances.Add(balance);
               }

               balance.AddTransaction(transactionType.Value, 1, msg.CorrelationId);
               logger.LogInformation(
                   "Processed attendance event: {CorrelationId} → {TransactionType} for student {StudentId}",
                   msg.CorrelationId, transactionType.Value, msg.StudentId);
           }

           // Upsert ProcessedAttendanceEvent
           if (processed is null)
           {
               dbContext.ProcessedAttendanceEvents.Add(new ProcessedAttendanceEvent
               {
                   CorrelationId = msg.CorrelationId,
                   LastStatus = newStatus,
                   UpdatedAt = DateTimeHelper.UtcNow()
               });
           }
           else
           {
               processed.LastStatus = newStatus;
               processed.UpdatedAt = DateTimeHelper.UtcNow();
           }

           await dbContext.SaveChangesAsync(context.CancellationToken);
       }
   }
   ```

   CRITICAL: The consumer uses `IgnoreQueryFilters()` because it runs in a background MassTransit consumer thread with NO HTTP context and NO `ITenantContext`. It queries by explicit `StudentId + SchoolId` from the message payload. This is the same pattern as `GetBySchoolAsync` in Organizations (Phase 02-01 decision).

   Query handler logic:
   ```csharp
   // Step 1: Find all processed events for this slot where LastStatus = "Present"
   var processedEvents = await dbContext.ProcessedAttendanceEvents
       .Where(p => p.LessonSlotId == query.LessonSlotId && p.LastStatus == "Present")
       .ToListAsync(cancellationToken);

   // Step 2: For each, find the AttendanceDebit transaction with matching CorrelationId
   var correlationIds = processedEvents.Select(p => p.CorrelationId).ToList();
   var debitTransactions = await dbContext.Set<LessonTransaction>()
       .IgnoreQueryFilters() // LessonTransaction has no filter, but include for safety
       .Where(t => correlationIds.Contains(t.CorrelationId!.Value)
                && t.Type == LessonTransactionType.AttendanceDebit)
       .ToListAsync(cancellationToken);

   // Step 3: For each debit, determine if Paid or Debt
   // Load the StudentBalance with all transactions ordered by Id (UUIDv7 = chronological)
   // For each debit transaction, compute running balance at that point:
   //   sum of all credit types with Id <= debit.Id minus sum of all debit types with Id < debit.Id
   //   If running balance before this debit was > 0 → "Paid", else → "Debt"
   var results = new List<StudentPaymentStatusDto>();
   foreach (var processed in processedEvents)
   {
       var debit = debitTransactions.FirstOrDefault(t => t.CorrelationId == processed.CorrelationId);
       if (debit is null) continue;

       // Get the student's balance to compute running total at debit time
       var balance = await dbContext.StudentBalances
           .IgnoreQueryFilters()
           .Include(b => b.Transactions)
           .FirstOrDefaultAsync(b => b.StudentId == processed.StudentId, cancellationToken);

       if (balance is null) continue;

       // Compute balance BEFORE this debit was applied
       var balanceBefore = balance.Transactions
           .Where(t => t.Id.CompareTo(debit.Id) < 0) // all transactions before this debit (UUIDv7 ordering)
           .Sum(t => t.Type is LessonTransactionType.ManualCredit or LessonTransactionType.AttendanceReversal
               ? t.Amount : -t.Amount);

       var status = balanceBefore > 0 ? "Paid" : "Debt";
       results.Add(new StudentPaymentStatusDto(processed.StudentId, status));
   }
   return results;
   ```

   ```csharp
   public sealed class PaymentsGrpcService(IMediator mediator)
       : Edvantix.Payments.Grpc.Generated.PaymentsGrpcService.PaymentsGrpcServiceBase
   {
       public override async Task<GetSlotPaymentStatusReply> GetSlotPaymentStatus(
           GetSlotPaymentStatusRequest request,
           ServerCallContext context)
       {
           var slotId = Guid.Parse(request.LessonSlotId);
           var statuses = await mediator.Send(
               new GetSlotPaymentStatusQuery(slotId),
               context.CancellationToken);
           var reply = new GetSlotPaymentStatusReply();
           reply.Statuses.AddRange(statuses.Select(s => new Generated.StudentPaymentStatus
           {
               StudentId = s.StudentId.ToString(),
               PaymentStatus = s.PaymentStatus
           }));
           return reply;
       }
   }
   ```
